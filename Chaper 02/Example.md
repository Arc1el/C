ex 2-4
```c
#include <stdio.h>
int main ()
{
	printf("%d 더하기 %d 는 %d 입니다.\n", 3, 5, 3+5);
	printf("%o 더하기 %i 는 %i 입니다.\n", 3, 5, 3+5);
	printf("%d - %d 는 %d 입니다.\n", 3, 5, 3+5);
	printf("%i - %i 는 %i 입니다.\n", 3, 5, 3+5);

	return 0;
}
```
